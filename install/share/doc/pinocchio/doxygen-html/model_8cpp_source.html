29  static Inertia convertFromUrdf(const ::urdf::Inertial & Y)
30  {
31  const ::urdf::Vector3 & p = Y.origin.position;
32  const ::urdf::Rotation & q = Y.origin.rotation;
33 
34  const Inertia::Vector3 com(p.x,p.y,p.z);
35  const Inertia::Matrix3 & R = Eigen::Quaterniond(q.w,q.x,q.y,q.z).matrix();
36 
37  Inertia::Matrix3 I;
38  I << Y.ixx,Y.ixy,Y.ixz,
39  Y.ixy,Y.iyy,Y.iyz,
40  Y.ixz,Y.iyz,Y.izz;
41  return Inertia(Y.mass,com,R*I*R.transpose());
42  }

65  void parseTree(::urdf::LinkConstSharedPtr link,
66  UrdfVisitorBase & model)
67  {
68  typedef UrdfVisitorBase::Scalar Scalar;
69  typedef UrdfVisitorBase::SE3 SE3;
70  typedef UrdfVisitorBase::Vector Vector;
71  typedef UrdfVisitorBase::Vector3 Vector3;
72  typedef Model::FrameIndex FrameIndex;
73 
74  // Parent joint of the current body
75  const ::urdf::JointConstSharedPtr joint =
76  ::urdf::const_pointer_cast< ::urdf::Joint>(link->parent_joint);
77 
78  if(joint) // if the link is not the root of the tree
79  {
80  PINOCCHIO_CHECK_INPUT_ARGUMENT(link->getParent());
81 
82  const std::string & joint_name = joint->name;
83  const std::string & link_name = link->name;
84  const std::string & parent_link_name = link->getParent()->name;
85  std::ostringstream joint_info;
86 
87  FrameIndex parentFrameId = getParentLinkFrame(link, model);
88 
89  // Transformation from the parent link to the joint origin
90  const SE3 jointPlacement
91  = convertFromUrdf(joint->parent_to_joint_origin_transform);
92 
93  const Inertia Y = convertFromUrdf(link->inertial);
94 
95  Vector max_effort(1), max_velocity(1), min_config(1), max_config(1);
96  Vector friction(Vector::Constant(1,0.)), damping(Vector::Constant(1,0.));
97  Vector3 axis (joint->axis.x, joint->axis.y, joint->axis.z);
98 
99  const Scalar infty = std::numeric_limits<Scalar>::infinity();
100 
101  switch(joint->type)
102  {
103  case ::urdf::Joint::FLOATING:
104  joint_info << "joint FreeFlyer";
105 
106  max_effort = Vector::Constant(6, infty);
107  max_velocity = Vector::Constant(6, infty);
108  min_config = Vector::Constant(7,-infty);
109  max_config = Vector::Constant(7, infty);
110  min_config.tail<4>().setConstant(-1.01);
111  max_config.tail<4>().setConstant( 1.01);
112 
113  friction = Vector::Constant(6, 0.);
114  damping = Vector::Constant(6, 0.);
115 
116  model.addJointAndBody(UrdfVisitorBase::FLOATING, axis,
117  parentFrameId,jointPlacement,joint->name,
118  Y,link->name,
119  max_effort,max_velocity,min_config,max_config,
120  friction,damping);
121  break;
122 
123  case ::urdf::Joint::REVOLUTE:
124  joint_info << "joint REVOLUTE with axis";
125 
126  // TODO I think the URDF standard forbids REVOLUTE with no limits.
127  assert(joint->limits);
128  if(joint->limits)
129  {
130  max_effort << joint->limits->effort;
131  max_velocity << joint->limits->velocity;
132  min_config << joint->limits->lower;
133  max_config << joint->limits->upper;
134  }
135 
136  if(joint->dynamics)
137  {
138  friction << joint->dynamics->friction;
139  damping << joint->dynamics->damping;
140  }
141 
142  model.addJointAndBody(UrdfVisitorBase::REVOLUTE, axis,
143  parentFrameId,jointPlacement,joint->name,
144  Y,link->name,
145  max_effort,max_velocity,min_config,max_config,
146  friction,damping);
147  break;
148 
149  case ::urdf::Joint::CONTINUOUS: // Revolute joint with no joint limits
150  joint_info << "joint CONTINUOUS with axis";
151 
152  min_config.resize(2);
153  max_config.resize(2);
154  min_config << -1.01, -1.01;
155  max_config << 1.01, 1.01;
156 
157  if(joint->limits)
158  {
159  max_effort << joint->limits->effort;
160  max_velocity << joint->limits->velocity;
161  }
162  else
163  {
164  max_effort << infty;
165  max_velocity << infty;
166  }
167 
168  if(joint->dynamics)
169  {
170  friction << joint->dynamics->friction;
171  damping << joint->dynamics->damping;
172  }
173 
174  model.addJointAndBody(UrdfVisitorBase::CONTINUOUS, axis,
175  parentFrameId,jointPlacement,joint->name,
176  Y,link->name,
177  max_effort,max_velocity,min_config,max_config,
178  friction,damping);
179  break;
180 
181  case ::urdf::Joint::PRISMATIC:
182  joint_info << "joint PRISMATIC with axis";
183 
184  // TODO I think the URDF standard forbids REVOLUTE with no limits.
185  assert(joint->limits);
186  if (joint->limits)
187  {
188  max_effort << joint->limits->effort;
189  max_velocity << joint->limits->velocity;
190  min_config << joint->limits->lower;
191  max_config << joint->limits->upper;
192  }
193 
194  if(joint->dynamics)
195  {
196  friction << joint->dynamics->friction;
197  damping << joint->dynamics->damping;
198  }
199 
200  model.addJointAndBody(UrdfVisitorBase::PRISMATIC, axis,
201  parentFrameId,jointPlacement,joint->name,
202  Y,link->name,
203  max_effort,max_velocity,min_config,max_config,
204  friction,damping);
205  break;
206 
207  case ::urdf::Joint::PLANAR:
208  joint_info << "joint PLANAR with normal axis along Z";
209 
210  max_effort = Vector::Constant(3, infty);
211  max_velocity = Vector::Constant(3, infty);
212  min_config = Vector::Constant(4,-infty);
213  max_config = Vector::Constant(4, infty);
214  min_config.tail<2>().setConstant(-1.01);
215  max_config.tail<2>().setConstant( 1.01);
216 
217  friction = Vector::Constant(3, 0.);
218  damping = Vector::Constant(3, 0.);
219 
220  model.addJointAndBody(UrdfVisitorBase::PLANAR, axis,
221  parentFrameId,jointPlacement,joint->name,
222  Y,link->name,
223  max_effort,max_velocity,min_config,max_config,
224  friction,damping);
225  break;
226 
227  case ::urdf::Joint::FIXED:
228  // In case of fixed joint, if link has inertial tag:
229  // -add the inertia of the link to his parent in the model
230  // Otherwise do nothing.
231  // In all cases:
232  // -let all the children become children of parent
233  // -inform the parser of the offset to apply
234  // -add fixed body in model to display it in gepetto-viewer
235 
236  joint_info << "fixed joint";
237  model.addFixedJointAndBody(parentFrameId, jointPlacement,
238  joint_name, Y, link_name);
239  break;
240 
241  default:
242  throw std::invalid_argument("The type of joint " + joint_name + " is not supported.");
243  break;
244  }
245 
246  model
247  << "Adding Body" << '\n'
248  << '\"' << link_name << "\" connected to \"" << parent_link_name << "\" through joint \"" << joint_name << "\"\n"
249  << "joint type: " << joint_info.str() << '\n'
250  << "joint placement:\n" << jointPlacement << '\n'
251  << "body info: " << '\n'
252  << " mass: " << Y.mass() << '\n'
253  << " lever: " << Y.lever().transpose() << '\n'
254  << " inertia elements (Ixx,Iyx,Iyy,Izx,Izy,Izz): " << Y.inertia().data().transpose() << '\n' << '\n';
255  }
256  else if (link->getParent())
257  throw std::invalid_argument(link->name + " - joint information missing.");
258 
259  BOOST_FOREACH(::urdf::LinkConstSharedPtr child, link->child_links)
260  {
261  parseTree(child, model);
262  }
263  }

272  void parseRootTree(const ::urdf::ModelInterface * urdfTree,
273  UrdfVisitorBase& model)
274  {
275  model.setName(urdfTree->getName());
276 
277  ::urdf::LinkConstSharedPtr root_link = urdfTree->getRoot();
278  model.addRootJoint(convertFromUrdf(root_link->inertial), root_link->name);
279 
280  BOOST_FOREACH(::urdf::LinkConstSharedPtr child, root_link->child_links)
281  {
282  parseTree(child, model);
283  }
284  }
